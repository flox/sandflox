   type CLIFlags struct {

   func ParseFlags(args []string) (*CLIFlags, []string) {
       flags := &CLIFlags{}
       fs := flag.NewFlagSet("sandflox", flag.ContinueOnError)
       fs.StringVar(&flags.Profile, "profile", "", "Override active profile")
       fs.StringVar(&flags.PolicyPath, "policy", "", "Path to policy.toml")

1. Create `main.go` with `package main`:

   ```go
   func main() {
       // 1. Parse CLI flags
       flags, userArgs := ParseFlags(os.Args[1:])

       // 2. Determine project directory
       //    If --policy specified, use its parent dir
       //    Otherwise, use current working directory (per D-08)
       projectDir := resolveProjectDir(flags)

       // 3. Find and parse policy.toml
       policyPath := filepath.Join(projectDir, "policy.toml")
       if flags.PolicyPath != "" {
           policyPath = flags.PolicyPath
       }
       policy, err := ParsePolicy(policyPath)
       // If no policy.toml, fall through to bare flox activate (matches bash fallback)
       // If parse error, print [sandflox] ERROR: and exit 1

       // 4. Resolve config (profile + merge + CLI overrides)
       config := ResolveConfig(policy, flags, projectDir)

       // 5. Write cache artifacts
       cacheDir := filepath.Join(projectDir, ".flox", "cache", "sandflox")
       WriteCache(cacheDir, config, projectDir)

       // 6. Emit diagnostics to stderr (per CORE-07)
       emitDiagnostics(config, flags.Debug)

       // 7. Exec into flox activate (per D-06, D-07)
       execFlox(userArgs)
   }
   ```

2. `resolveProjectDir(flags *CLIFlags) string`:
   - If `flags.PolicyPath != ""`, use `filepath.Dir(flags.PolicyPath)` as project dir
   - Otherwise `os.Getwd()` (per D-08: binary assumes running from project directory)

3. `emitDiagnostics(config *ResolvedConfig, debug bool)`:
   - Always print: `[sandflox] Profile: {profile} | Network: {netMode} | Filesystem: {fsMode}` to stderr (matches existing bash format at sandflox.bash line 139)
   - If debug: also print requisites file, allow-localhost, writable paths, denied paths, cache dir (per CORE-07)
   - Use `fmt.Fprintf(os.Stderr, ...)` -- not `log` package (per anti-pattern in RESEARCH.md)
   - Accept an `io.Writer` parameter (or use a package-level `var stderr io.Writer = os.Stderr`) so that `main_test.go` can capture output without subprocess spawning

4. `execFlox(userArgs []string)`:
   - `exec.LookPath("flox")` to find flox binary; if not found: `[sandflox] ERROR: flox not found in PATH` and exit 1
   - Build argv: if `len(userArgs) == 0` -> `["flox", "activate"]` (interactive per D-07)
   - If `len(userArgs) > 0` -> `["flox", "activate", "--", userArgs...]` (non-interactive per D-07)
   - Call `syscall.Exec(floxPath, argv, os.Environ())` for clean process replacement (per D-09, CORE-01 via syscall.Exec not os/exec.Command)
   - If syscall.Exec fails (should not return on success), print error and exit 1

5. Handle the no-policy fallback: if `policy.toml` does not exist, print `[sandflox] No policy.toml found -- falling back to flox activate` to stderr and exec directly into flox activate (matches sandflox.bash line 25-26).

6. Add a `var Version = "dev"` at package level for build-time injection via ldflags (used by Nix build in Plan 03).
